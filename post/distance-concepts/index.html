Euclidean diantance

2d euclidean distance

$$ d{12} =\sqrt{(x{1}-x{2})^2+(y{1}-y_{2})^2} $$

from numpy import *

def twoPointDistance(a,b):
    d = sqrt( (a[0]-b[0])**2 + (a[1]-b[1])**2 )
    return d

3d Euclidean distance

$$ d{12} =\sqrt{(x{1}-x{2})^2+(y{1}-y{2})^2+(z{1}-z_{2})^2} $$

def threePointDistance(a,b):
    d = sqrt( (a[0]-b[0])**2 + (a[1]-b[1])**2 + (a[2]-b[2])**2 )
    return d

N-dimensional Euclidean distance

$$ \sqrt{\sum{n}^{k=1}(x{1k}-x_{2k})^2 } $$

def distance(a,b):
    sum = 0
    for i in range(len(a)):
        sum += (a[i]-b[i])**2
    return sqrt(sum)

Weighted Euclidean distance

$$d{12} =\sqrt {\sum{k=1}^{n} (\frac{x{1k}-x{2k}}{s_{k}})^2}$$

def moreBZOSdis(a,b):
    sumnum = 0
    for i in range(len(a)):
        # calculating si standard deviation
        avg = (a[i]-b[i])/2
        si = sqrt( (a[i] - avg) ** 2 + (b[i] - avg) ** 2 )
        sumnum += ((a[i]-b[i])/si ) ** 2

    return sqrt(sumnum)

Manhattan distance

$$d{12} = \sum{k=1}^{n} \left | x{1k} - x{2k} \right |$$

def moreMHDdis(a,b):
    sum = 0 
    for i in range(len(a)):
        sum += abs(a[i]-b[i])
    return sum

Chebyshev Distance

2d:

$$d{12} = max( \left | x{1} - x{2} \right | , \left | y{1} - y_{2} \right |)$$

def twoQBXFdis(a,b):
    return max( abs(a[0]-b[0]), abs(a[1]-b[1]))

n-dimensional:

$$d12 = max{i\epsilon n}( \left | x{1i} - x_{2i} \right | )$$

def moreQBXFdis(a,b):
    maxnum = 0
    for i in range(len(a)):
        if abs(a[i]-b[i]) > maxnum:
            maxnum = abs(a[i]-b[i])
    return maxnum

N-dimensional cosine similarity between two vectors:

$$\cos \theta = \frac{ \sum{k=1}^{n} x{1k}x{2k} }{ \sqrt{ \sum{k=1}^{n}x{1k}^2 }\sqrt{ \sum{k=1}^{n} x_{2k}^2 } }â€‹$$

where $\cos \theta$ range from -1 to 1.

def moreCos(a,b):
    sum_fenzi = 0.0
    sum_fenmu_1,sum_fenmu_2 = 0,0
    for i in range(len(a)):
        sum_fenzi += a[i]*b[i]
        sum_fenmu_1 += a[i]**2 
        sum_fenmu_2 += b[i]**2 

    return sum_fenzi/( sqrt(sum_fenmu_1) * sqrt(sum_fenmu_2) )
